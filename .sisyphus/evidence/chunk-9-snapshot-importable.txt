## Snapshot schema check

Working directory: `/opt/flowmanner/backend`

Command:
```bash
python3 - <<'PY'
import importlib.util
from pathlib import Path
from app.models import Base

spec = importlib.util.spec_from_file_location('snapshot_model_metadata', Path('scripts/snapshot_model_metadata.py'))
module = importlib.util.module_from_spec(spec)
spec.loader.exec_module(module)

snapshot = module.build_snapshot(Base.metadata)
assert list(snapshot.keys()) == ['generated_at', 'alembic_version', 'model_count', 'tables']
assert snapshot['generated_at'].endswith('Z')
assert snapshot['alembic_version'] == ''
assert snapshot['model_count'] == len(snapshot['tables'])

for table in snapshot['tables']:
    assert set(table.keys()) == {'name', 'columns', 'indexes', 'unique_constraints', 'foreign_keys'}
    assert isinstance(table['columns'], dict)
    assert list(table['columns'].values()) == [table['columns'][name] for name in sorted(table['columns'])]
    assert all(isinstance(column_type, str) for column_type in table['columns'].values())
    assert table['indexes'] == sorted(table['indexes'])
    assert all(isinstance(index_name, str) for index_name in table['indexes'])
    assert table['unique_constraints'] == sorted(table['unique_constraints'])
    assert all(isinstance(columns, list) for columns in table['unique_constraints'])
    assert table['foreign_keys'] == sorted(table['foreign_keys'])
    assert all(isinstance(pair, list) and len(pair) == 2 for pair in table['foreign_keys'])

print(f"schema ok: {snapshot['model_count']} tables")
PY
```
